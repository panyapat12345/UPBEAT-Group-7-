m = 0
i = 10
Budget = 50

while (i) {
    Random = 100 + 101*i^4
    dir = Random % 6

    if (dir - 4) then move upleft
    else if (dir - 3) then move downleft
    else if (dir - 2) then move down
    else if (dir - 1) then move downright
    else if (dir) then move upright
    else move up
    m = m + 1

    i = i - 1
}
